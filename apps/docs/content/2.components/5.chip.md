## Removable

::ChipContent{:removable="true"}
::

::code-collapse
```vue
<template>
  <div class="flex flex-wrap gap-3">
    <Chip 
      v-for="chip in chips" 
      :key="chip.label" 
      :label="chip.label" 
      :severity="chip.severity" 
      removable 
      @remove="removeChip(chip)" 
    />
  </div>
</template>

<script setup lang="ts">
import { Chip } from "veloce-vue/ui";
import { ref } from "vue";

const chips = ref([
  { label: "Apple", severity: "primary" as const },
  { label: "Facebook", severity: "info" as const },
  { label: "Google", severity: "success" as const },
  { label: "Microsoft", severity: "warning" as const },
  { label: "GitHub", severity: "error" as const },
]);

const removeChip = (chip: { label: string; severity: string }) => {
  chips.value = chips.value.filter((c) => c.label !== chip.label);
};
</script>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `label` | `string` | `""` | Label text displayed in the chip |
| `severity` | `"primary" \| "secondary" \| "success" \| "info" \| "warning" \| "error" \| "neutral"` | `"primary"` | Color severity of the chip |
| `size` | `"sm" \| "md" \| "lg" \| "xl"` | `"md"` | Size of the chip |
| `icon` | `Component` | `null` | Icon component to display before the label |
| `removable` | `boolean` | `false` | Show remove button and emit remove event |
| `rounded` | `boolean` | `false` | Apply full border radius (pill shape) |

## Events

| Event | Description |
|-------|-------------|
| `remove` | Emitted when the remove button is clicked |
