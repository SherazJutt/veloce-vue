import { Modal } from "veloce/ui";

## Basic

::ModalContent{:basic="true"}
::

::code-collapse
  ```vue
  <template>
    <Modal v-model="show">
      <Button label="Open Modal" />
      <template #content>
        <p class="text-muted">This is the modal content. You can add any content here.</p>
      </template>
    </Modal>
  </template>
  
  <script setup lang="ts">
  import { Modal, Button } from "veloce-vue/ui";
  import { ref } from "vue";
  
  const show = ref(false);
  </script>
  ```
::

## With Footer

::ModalContent{:withFooter="true"}
::

::code-collapse
```vue
<template>
  <Modal v-model="show">
    <Button label="Open Modal with Footer" />
    <template #content>
      <p class="text-muted">Modal content goes here.</p>
    </template>
    <template #footer>
      <div class="flex justify-end gap-2">
        <Button label="Cancel" variant="soft" @click="show = false" />
        <Button label="Save" @click="show = false" />
      </div>
    </template>
  </Modal>
</template>

<script setup lang="ts">
import { Modal, Button } from "veloce-vue/ui";
import { ref } from "vue";

const show = ref(false);
</script>
```
::

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `v-model` | `boolean` | `false` | Two-way binding for modal visibility |

## Slots

| Slot | Description |
|------|-------------|
| `default` | Trigger element that opens the modal when clicked |
| `content` | Main content area of the modal |
| `footer` | Footer section (optional) |
